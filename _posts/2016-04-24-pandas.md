**DataFrame**
```
data = {
  'A': [1, 2, 3],
  'B': ['a', 'b', 'c'],
  'C': [1.1, 2.1, 3.1]
}

df = DataFrame(data)

# インデックスのタイトル
df.index.name = "bar"

# カラムのタイトル
df.column.name = "bar"

# データフレームのカラムの順番を指定する
df = DataFrame(data, columns=['C', 'A', 'B'], index=['one', 'two', 'three'])

# 上記と同じ
data = {
  'A': {'one':1, 'two':2, 'three':3},
  'B': {'one':'a', 'two':'b', 'three':'c'},
  'C': {'one':1.1, 'two':2.1, 'three':3.1}
}
df = DataFrame(data)

# 転置
df.T

# 2次元のnarrayを返す
df.values

# データが無い場合はNaNが割り当てられる
# D列とfour行の値はNaN
df = DataFrame(data, columns=['C', 'A', 'B', 'D'], index=['one', 'two', 'three', 'four'])

# カラム名の一覧を取得
df.columns

# 列をSeriesとして取得
df['A']
df.A

# 行をSeriesとして取得
df.ix['one']

# 値をセット
df['C'] = 1.5
df['C'] = numpy.arange(4)
s = Series([2.9, 3.9], index=['two', 'three'])
# 無いものはNaN
df['C'] = s
# 列の削除
del df['C']
```
