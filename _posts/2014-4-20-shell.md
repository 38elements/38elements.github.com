`'`と`"`で囲まれた文字列は展開しない   
<br>

```
foo=bar
echo $foo
// => bar
echo ${foo}hoge
// => barhoge
echo ${a:-b}
// => b
a=c
echo ${a:-b}
// => c
a=
echo ${a:=b}
// => b
echo ${a:=c}
// => b
echo ${a:+c}
// => c
a=
echo ${a:+c}
// =>
a=b
echo ${a:?ABC}
// => b
a=
echo ${a:?ABC}
ABC
// 後ろから文字列を取り除く
a=1a2a3
echo ${a%a*}
// => 1a2
echo ${a%%a*}
// => 3
// 前から文字を取り除く
echo ${a#*a}
// => 2a3
echo ${a##*a}
// => 3
```
