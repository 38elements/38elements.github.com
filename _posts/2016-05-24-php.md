class Car extends Base
{
    use FooTrait, BarTrait;
    
    const NAME = 'name';
    
    public $bar;
    public static $car_static = 'foo';

    {
        parent::__construct($foo);
        $this->bar = function() {
            return 42;
        };
    }
    
    public function foo()
    {
        echo self::$car_static;
    }
    
    // サブクラスでオーバーライドされない
    final public function foo()
    {
        echo self::$car_static;
    }
}
